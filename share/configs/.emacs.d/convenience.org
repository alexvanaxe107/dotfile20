#+title AVA Convenient Commands

* Font Manipulation Commands

** Fix font size

#+begin_src emacs-lisp

  (defun fixfont(size)
  (let
      ((font (concat (get-font) size)))
      (set-frame-parameter (selected-frame) 'font font)
  ))

#+end_src
